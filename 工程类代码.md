# 2. 工具节点 → 独立 Agent

```python
# 当前：只是函数
async def cypher_query(state):
    return {"result": ...}

# 改成：独立 Agent
class CypherAgent:
    def __init__(self, llm, memory, tools):
        self.llm = llm
        self.memory = memory  # 每个 Agent 有自己的记忆
        self.tools = [schema_retriever, cypher_executor]

    async def run(self, task):
        # Agent 自己决定用哪个工具、问什么问题
        thought = await self.llm.think(task, self.memory)
        action = await self.select_tool(thought)
        result = await self.execute(action)
        await self.memory.add(task, result)
        return result
```

# 3. Agent 间通信（工具到多agent合作）

```python
# 当前：统一汇总到 Summarize
summarize.add_edge("cypher_query", "summarize")
summarize.add_edge("predefined_cypher", "summarize")

# 改成：Agent 可以互相通信
class Agent:
    async def run(self, task):
        if "需要schema" in task:
            schema = await self.send_to("SchemaAgent", "给我实体类型")
            task = f"{task} [上下文: {schema}]"

        result = await self.execute(task)

        # 可以继续委托给其他 Agent
        if self.should_delegate(result):
            return await self.send_to("WriterAgent", result)
        return result
```

# 4.实现一个简单的multi-agent

```python
# 1. 保留 Guardrails 作为边界 Agent
class GuardrailsAgent(Agent):
    system_prompt = "你是安全护栏，决定问题是否在知识图谱范围内"
    tools = [block_tool, allow_tool]

# 2. Planner 变成 PlannerAgent
class PlannerAgent(Agent):
    system_prompt = "你是任务规划师，将复杂问题分解为子任务"
    tools = []  # 只负责分解，不执行

    async def run(self, task):
        subtasks = await self.llm.decompose(task)
        for subtask in subtasks:
            await self.delegate_to(subtask, ["RetrieverAgent", "SQLAgent"])
        return await self.collect_results()

# 3. 各专业 Agent
class RetrieverAgent(Agent):
    system_prompt = "你是检索专家，负责从向量数据库检索相关内容"
    tools = [pgvector_tool, milvus_tool, neo4j_tool]

    async def run(self, task):
        # Agent 自己决定用哪个工具
        decision = await self.llm.reason(task)
        if decision.need_structured:
            return await self.use_tool("pgvector", task)
        else:
            return await self.use_tool("milvus", task)

class SQLAgent(Agent):
    system_prompt = "你是 SQL 专家，将自然语言转为 SQL 查询"
    tools = [mysql_tool, postgresql_tool]

# 4. WriterAgent (原 Summarize)
class WriterAgent(Agent):
    system_prompt = "你是写作助手，汇总各 Agent 结果生成最终答案"
    tools = []
```

## 4.**Message Passing 消息传递（企业级多 Agent）**

Message Passing = 让多个 Agent 像发邮件一样协作

不直接见面，不互相依赖，不共享记忆，大规模分布式协作。

如A2A，ACA

## 5. **Tool Call 工具调用（OpenAI 官方标准）**

把其他 Agent **当成工具（Tool）**，

LLM 自己决定什么时候调用。

```python
<|tool_call|> name="WriterAgent" params={"text": "..."}
```
